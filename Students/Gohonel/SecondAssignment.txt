public class SecondAssignment {
	public static void main(String[] args) 
	{
		int i = 1 , j = 2, sum = 0;
		
		while ( j < 4000000 )
		{
			if ( (j % 2 == 0) && (j < 4000001) )
			{
				sum = sum + j;
			}
			
			j = j + i; 
			i = j - i;
		}
		System.out.print("Your result is: ");
		System.out.println(sum);
	}
}
